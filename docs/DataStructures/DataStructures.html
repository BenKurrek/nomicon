Transaction

 
struct Transaction {
    originator: AccountId,
    receiver: AccountId,
    nonce: u64,
    method_name: Vec<u8>,
    args: Vec<u8>,
    amount: Balance,
}

struct ReceiptBlock {
    header: SignedShardBlockHeader,
    path: Vec<Hash>,

Shard

 
struct ShardBlockHeader {
    parent: Hash,
    index: u64,
    shard_id: u64,
    // State trie
    merkle_root_state: Hash,
    // Transactions Trie
    merkle_root_tx: Hash,
    // Receipt blocks Trie
    merkle_root_receipt: Hash,
    // Transaction result Trie
    merkle_root_tx_result: Hash,
    // Receipt result Trie
    merkle_root_receipt_result: Hash,
}

struct tShardBlockBody {
    receipts: Vec<ReceiptBlock>,
    transactions: Vec<SignedTransaction>,
}

struct SignedShardBlock {
    header: ShardHeader,
    body: ShardBody,
    signature: BLS,
    authority_mask: Vec<bool>,
}
